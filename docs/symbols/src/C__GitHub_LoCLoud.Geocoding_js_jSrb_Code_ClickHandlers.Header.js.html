1 function handlerBtnUpgrade() {
  2     showMsgBox("Please wait while upgrade commences. This operation may take several minutes");
  3     jQuery.getJSON("./ws/ws-upgrade.php", {
  4         pVersion: jQuery("#btnUpgrade").data("version")
  5     }, function(pData) {
  6         if (pData.v === 1) {
  7             showMsgBox(pData.m, false);
  8             jQuery("#btnUpgrade").hide();
  9         } else {
 10             showMsgBox(pData.m, true);
 11         }
 12     }).fail(function() {
 13         showMsgBox("Web service request failed", true);
 14     });
 15 }

 17 function handlerBtnHelp() {

 19         url: "./res/help.html"
 20     }).done(function(data) {
 21         jQuery("#dlgHelp").html(data);
 22     });
 23     jQuery("#dlgHelp").dialog(
 24             {

 33                 hide: "fade",
 34                 show: "fade",

 36                 modal: true,
