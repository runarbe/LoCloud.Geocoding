1 /**
  2  * Set the currently visible tab in the new datasource wizard dialogs
  3  * 
  4  * @param {Number} pEnabledTab A number between 1 and 3
  5  * @returns {void}
  6  */
  7 function SetNewDataSourceTab(pEnabledTab) {
  8     var mDisabledTabs = [0,1,2];
  9     var mTabToRemove = (mDisabledTabs.indexOf(pEnabledTab));
 10     mDisabledTabs.splice(mTabToRemove, 1);
 11     
 12     ctlNewDataSourceWizard.tabs("option", {
 13         disabled: mDisabledTabs,
 14         active: pEnabledTab,
 15         heightStyle: "fill"
 16     });
 17 }
