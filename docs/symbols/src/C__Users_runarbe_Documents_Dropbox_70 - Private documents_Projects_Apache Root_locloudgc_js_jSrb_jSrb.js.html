1 /**
  2  * The name of the root script
  3  * @type String
  4  */
  5 var scriptName = "jSrb.js";
  6 
  7 /**
  8  * @namespace This is the toplevel namespace for the LoCloud geocoding micro
  9  * service
 10  */
 11 var jSrb = {
 12     /**
 13      * Internal function to get the location of the currently executed
 14      * script
 15      */
 16     _getScriptLocation: function() {
 17         var r = new RegExp("(^|(.*?\\/))(" + scriptName + ")(\\?|$)"),
 18                 s = document.getElementsByTagName('script'),
 19                 src, m, l = "";
 20         for (var i = 0, len = s.length; i < len; i++) {
 21             src = s[i].getAttribute('src');
 22             if (src) {
 23                 m = src.match(r);
 24                 if (m) {
 25                     l = m[1];
 26                     break;
 27                 }
 28             }
 29         }
 30         return l;
 31     },
 32     /**
 33      * @namespace This namespace provides functions for initializing the
 34      * application, laying out elements and attaching handlers to intereactive
 35      * controls
 36      */
 37     AppSetup: {},
 38     /**
 39      * @namespace Utility functions
 40      */
 41     Utilities: {},
 42     /**
 43      * @namespace Interactive elements handler functions
 44      */
 45     Handlers: {},
 46     /**
 47      * @namespace Map functions
 48      */
 49     Map: {},
 50     /**
 51      * @namespace Configuration values
 52      */
 53     Configuration: {}
 54 };
 55 
 56 /**
 57  * Array of JavaScript files to be included into the main script
 58  * @type Array
 59  */
 60 jSrb.jsFiles = [
 61     "Code/JavaScriptClasses.js",

 63     "Code/Utility.Map.js",

 72     "Code/ClickHandlers.Header.js",
 73     "Code/ClickHandlers.NewDatasource.js",
 74     "Code/MapLayers.js",
 75     "Code/Main.js" //Always at end
 76 ];
 77 
 78 jSrb.scriptTags = new Array(jSrb.jsFiles.length);
 79 jSrb.host = jSrb._getScriptLocation();
 80 for (var i = 0, len = jSrb.jsFiles.length; i < len; i++) {
 81     jSrb.scriptTags[i] = "<script src='" + jSrb.host + jSrb.jsFiles[i] +
 82             "'></script>";
 83 }
 84 if (jSrb.scriptTags.length > 0) {
 85     document.write(jSrb.scriptTags.join(""));
