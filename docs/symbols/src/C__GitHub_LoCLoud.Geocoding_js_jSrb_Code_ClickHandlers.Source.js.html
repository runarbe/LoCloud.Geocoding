 88 /**
 89  * Function that is triggered when the users selects a source item from the
 90  * item list on the bottom right hand side of the screen
 91  * 
 92  * @returns {void}
 93  */
 94 function handlerSelectSourceItem() {
 95     jQuery("#selectable li").removeClass("ui-state-highlight");
 96     getSelectedSourceItem().addClass("ui-state-highlight")
 97             .each(function() {
 98         var data = getSelectedSourceItem().data("attributes");
 99         //console.log(data);
100 
101         jQuery("#tbItemName").val(data._nc);
102         jQuery("#hdnTableId").val(data._id);
103         jQuery("#hdnTableName").val(mDatasource.ds_table);
104 
105         jQuery("#hdnFieldChanges").val(data.gc_fieldchanges);
106         
107         jQuery("#tbLongitude").val(null);
108         jQuery("#tbLatitude").val(null);
109 
110         // Initialize button to show URL
111         if (data._uc === null) {
112             jQuery("#btnViewUrl").hide();
113         } else {
114             jQuery("#hdnUrl").val(data._uc);
115             jQuery("#btnViewUrl").show();
116         }
117 
118         // Initialize button to show image
119         if (data._ic === null) {
120             jQuery("#btnViewImage").hide();
121         } else {
122             jQuery("#hdnImage").val(data._ic);
123             jQuery("#btnViewImage").show();
124         }
125 
126         clearIcons();
127 
128         if (data.gc_probability != null) {
129             jQuery("#radio input:radio[value=" + data.gc_probability + "]").prop("checked", true).button("refresh");
130         }
131 
132         /*
133          * Remove alternate markers from map
134          */
135         jQuery.each(alternateMarkers, function(key2, val2) {
136             markers.removeMarker(val2);
137         });
138 
139         /*
140          * Add alternate markers to map
141          */
142         jQuery.each(data.gc_alternates, function(key3, val3) {
143             var mLonLat = new OpenLayers.LonLat(val3.gc_lon, val3.gc_lat).transform(p4326, p900913);
144             addAlternateIcons(mLonLat.clone());
145             mLonLat = null;
146         });
147 
148         if (data.gc_lon != null && data.gc_lat != null) {
149             var mLonLat2 = new OpenLayers.LonLat(data.gc_lon, data.gc_lat).transform(p4326, pDatasource);
150             jQuery("#tbLongitude").val(roundCoordinates(mLonLat2.lon, mDatasource.ds_coord_prec));
151             jQuery("#tbLatitude").val(roundCoordinates(mLonLat2.lat, mDatasource.ds_coord_prec));
152 
153             var mLonLat = new OpenLayers.LonLat(data.gc_lon, data.gc_lat).transform(p4326, p900913);
154             addProposedIcon(mLonLat.clone());
155             map.setCenter(mLonLat, defaultZoomTo);
156 
157         }
158 
159         if (data._x != null && data._y != null) {
160             //console.log('add original coordinates');
161             mLonLat = new OpenLayers.LonLat(data._x, data._y).transform(pDatasource, p900913);
162             addExistingIcon(mLonLat.clone());
163 
164             if (jQuery("#tbLongitude").val() == "" || (data.gc_lon == null || data.gc_lat == null)) {
165                 jQuery("#tbLongitude").val(data._x);
166                 jQuery("#tbLatitude").val(data._y);
167                 map.setCenter(mLonLat, defaultZoomTo);
168             }
169 
170         }
171 
172         jQuery(".ui-selected", "#selectable").first().effect("transfer", {
173             to: jQuery("#divForm")
174         }, 500);
175 
176         showGeocodingForm();
177 
178     });
179 }

181 /**
182  * Function that is triggered when a user selects a datasource in the datasource
183  * panel
184  * 
185  * @returns {void}
186  */
187 function handlerSelectDatasource() {
188 
189     /*
190      * Hide datasource filters
191      */
192     jQuery("#frmFilterByArea").hide();
193     jQuery("#sbFilterAdm0").val(null).hide();
194     jQuery("#sbFilterAdm1").val(null).hide();
195     jQuery("#frmFilterByCategory").val(null).hide();
196 
197     clearDatasourceFilters();
198 
199     /*
200      * Get the selected value of the datasource dropdown
201      */
202     var mDatasourceVal = jQuery("#sbDatasource option:selected").val();
203 
204     /*
205      * If a value is selected, proceed
206      */
207     if (mDatasourceVal != "") {
208 
209         /*
210          * Assign datasource to global variable
211          */
212         mDatasource = getSelectedDatasource();
213         pDatasource = new OpenLayers.Projection("EPSG:" + mDatasource.ds_srs);
214         listSrcDataStartItem = 0;
215 
216         if (mDatasource.ds_col_adm0 != null && mDatasource.ds_col_adm0 != "") {
217             loadAdm0(mDatasourceVal);
218         }
219 
220         if (mDatasource.ds_col_cat != null && mDatasource.ds_col_cat != "") {
221             //console.log(mDatasource.ds_col_cat)
222             loadCategories(mDatasourceVal);
223         }
224 
225         loadSourceItems2(listSrcDataStartItem, listSrcDataNumItems);
226         clearIcons();
227         resetGeocodingForm();
228         hideGeocodingForm();
229     } else {
230         jQuery("#selectable").empty();
231         clearDatasourceFilters();
232         pDatasource = null;
233         hideGeocodingForm();
234     }
235 }
