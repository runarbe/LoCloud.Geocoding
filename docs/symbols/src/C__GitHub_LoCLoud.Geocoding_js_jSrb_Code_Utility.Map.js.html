1 function addExistingIcon(pLonLat) {
  2     if (currentMarker != null) {
  3         markers.removeMarker(currentMarker);
  4     }
  5     var size = new OpenLayers.Size(27,27);
  6     var offset = new OpenLayers.Pixel(-(size.w/2), -size.h);
  7     var icon = new OpenLayers.Icon('images/white01.png',size,offset);
  8     currentMarker = new OpenLayers.Marker(pLonLat,icon);
  9     markers.addMarker(currentMarker);    
 10 }
