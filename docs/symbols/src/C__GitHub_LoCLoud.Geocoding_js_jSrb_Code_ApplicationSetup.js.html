 29     jSrb.AppSetup.reflow = function() {
 30         var appWidth = jQuery(window).width();
 31         var appHeight = jQuery(window).height();
 32         var hdrHeight = 100;
 33         var stbHeight = 30;
 34         var srcHeight = appHeight - (hdrHeight + stbHeight);
 35         var srcWidth = 250;
 36         var schHeight = appHeight - (hdrHeight + stbHeight);
 37         var schWidth = 250;
 38         var frmHeight = 165;
 39         var frmBottom = stbHeight;
 40         var mapWidth = (appWidth - (srcWidth + schWidth)) - 4;
 41         var mapHeight = (appHeight - (hdrHeight + stbHeight + frmHeight)) - 4;
 42         jQuery("#divHeader").css("width", appWidth).css("height", hdrHeight);
 43         jQuery("#divSource").css("width", srcWidth).css("height", srcHeight).css("top", hdrHeight).css("left", 0);
 44         jQuery("#divSearch").css("width", schWidth).css("height", schHeight).css("top", hdrHeight).css("right", 0);
 45         jQuery("#divMap").css("width", mapWidth).css("height", mapHeight).css("top", hdrHeight).css("left", srcWidth);
 46         jQuery("#theMap").css("width", mapWidth).css("height", mapHeight);
 47         jQuery("#divStatusBar").css("width", appWidth).css("height", stbHeight).css("bottom", 0).css("left", 0);
 48         jQuery("#divForm").css("width", mapWidth).css("height", frmHeight).css("bottom", frmBottom).css("left", srcWidth);
 49     };

500     /**
501      * Show the upgrade button on the header menu if an update is available for
502      * download. This function is only available for users with a elevated access
503      * privileges
504      */
505     jSrb.AppSetup.checkUpgrade = function() {
506         jQuery.getJSON(WsUrl.getLatestVersion, null, function(pData) {
507             if (pData.v === -1) {
508                 jQuery("#btnUpgrade").show().button('option', 'label', 'Upgrade to v' + pData.d);
509                 jQuery("#btnUpgrade").data("version", pData.d[0]);
510             }
511         });
512     };
