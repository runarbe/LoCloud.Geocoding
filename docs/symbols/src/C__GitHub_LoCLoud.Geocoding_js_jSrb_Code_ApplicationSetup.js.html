 23     /**
 24      * Setup, place and size window panes
 25      *
 26      * @returns void
 27      */
 28     function reflow() {
 29         var appWidth = jQuery(window).width();
 30         var appHeight = jQuery(window).height();
 31         var hdrHeight = 100;
 32         var stbHeight = 30;
 33         var srcHeight = appHeight - (hdrHeight + stbHeight);
 34         var srcWidth = 250;
 35         var schHeight = appHeight - (hdrHeight + stbHeight);
 36         var schWidth = 250;
 37         var frmHeight = 165;
 38         var frmBottom = stbHeight;
 39         var mapWidth = appWidth - (srcWidth + schWidth);
 40         var mapHeight = appHeight - (hdrHeight + stbHeight + frmHeight);
 41         jQuery("#divHeader").css("width", appWidth).css("height", hdrHeight);
 42         jQuery("#divSource").css("width", srcWidth).css("height", srcHeight).css("top", hdrHeight).css("left", 0);
 43         jQuery("#divSearch").css("width", schWidth).css("height", schHeight).css("top", hdrHeight).css("right", 0);
 44         jQuery("#divMap").css("width", mapWidth).css("height", mapHeight).css("top", hdrHeight).css("left", srcWidth);
 45         jQuery("#theMap").css("width", mapWidth).css("height", mapHeight);
 46         jQuery("#divStatusBar").css("width", appWidth).css("height", stbHeight).css("bottom", 0).css("left", 0);
 47         jQuery("#divForm").css("width", mapWidth).css("height", frmHeight).css("bottom", frmBottom).css("left", srcWidth);
 48     }

488     /**
489      * Show the upgrade button on the header menu if an update is available for
490      * download. This function is only available for users with a elevated access
491      * privileges
492      * 
493      * @return void
494      */
495     function checkUpgrade() {
496         jQuery.getJSON("./ws/ws-check-updates.php", null, function(pData) {
497             if (pData.v === -1) {
498                 jQuery("#btnUpgrade").show().button('option', 'label', 'Upgrade to v' + pData.d);
499                 jQuery("#btnUpgrade").data("version", pData.d[0]);
500             }
501         });
502     }
